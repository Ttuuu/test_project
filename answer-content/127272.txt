<pre><code>public static int[] maxRunOfPositiveNumbers(int[] arr) {
  final int n = arr.length;
  int maxstart = 0;
  int maxlen = 0;
  int i = 0;

  while (i &lt; n) {
    while (i &lt; n &amp;&amp; arr[i] &lt;= 0)
      i++;

    int start = i;
    while (i &lt; n &amp;&amp; arr[i] &gt; 0)
      i++;
    int len = i - start;

    if (len &gt; maxlen) {
      maxstart = start;
      maxlen = len;
    }
  }
  return Arrays.copyOfRange(arr, maxstart, maxstart + maxlen);
}
